1- chown -R postgres /home/username/misdb

2- chgrp -R postgres /home/username/misdb

3- psql -U postgres

4- CREATE TABLESPACE tbl_misdb LOCATION '/home/username/misdb';

1- CREATE DATABASE dbejemplo TABLESPACE = tbl_misdb;
